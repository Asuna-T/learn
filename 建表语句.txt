DROP TABLE IF EXISTS student;

CREATE TABLE student
(
   sno         INT             auto_increment,
   name        VARCHAR(200)    NULL,
   age         INT             NULL,
    PRIMARY KEY(sno)
) CHARACTER SET = utf8;
